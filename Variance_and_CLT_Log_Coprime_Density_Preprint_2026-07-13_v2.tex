\begin{abstract}
Let $X_1, X_2, \ldots$ be independent and identically distributed random variables uniform on $\{2, 3, \ldots, M\}$, let $L_k = \lcm(X_1, \ldots, X_k)$, and let
\[
W_k(M) := \ln(\varphi(L_k)/L_k) - \EE\,\ln(\varphi(L_k)/L_k)
\]
denote the centred log coprime density, with $s_k^2 := \Var W_k(M)$. A companion paper~\cite{wakid1} established an exact subset identity for $\rho_k(M) = \EE[\varphi(L_k)/L_k]$, a sharp asymptotic equivalence $\rho_k(M) \to P_M$ at fixed $M$, and the joint-regime convergence $\ln M \cdot (\rho_{\lfloor cM \rfloor}(M) - P_M)/P_M \to A(c)$ with $A(c) := \sum_{j \geq 1} e^{-cj}\ln(1 + 1/j)$. We study second-order fluctuations in the same joint regime $k = \lfloor cM \rfloor$, $M \to \infty$, with fixed $c > 0$. We prove two results. First, the variance asymptotic
\[
M \ln M \cdot s_k^2 \longrightarrow B(c) := \frac{1}{e^c - 1} - \frac{1}{e^{2c} - 1},
\]
via a diagonal / off-diagonal decomposition where the off-diagonal term is shown to be negligible by an exact covariance identity valid for prime pairs $(p, q)$ with $pq > M$. Second, a central limit theorem: $W_k(M)/s_k \xrightarrow{d} \NN(0, 1)$ as $M \to \infty$ in the joint regime. The CLT proof uses the Hall--Heyde martingale central limit theorem applied to the Doob filtration $\F_i = \sigma(X_1, \ldots, X_i)$. Two ingredients enter. A uniform increment bound $\max_i |D_i| = O((\log M)^3/M) = o(s_k)$ on a super-polynomially likely event combines a coupon-collector tail estimate with a sieve-geometric observation ($(M/(\log M)^2)^2 > M$ for $M$ large), either fact alone being insufficient. Conditional-variance concentration is obtained from explicit moment bounds for the hitting-time variables $T_p := \min\{i : p \mid X_i\}$: a per-prime variance bound $\Var(R_{pp}^{(k)}) \leq 3 q_p^k/\pi_p^2$ and a cross-prime covariance bound $|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})| \leq C_0(c) k q_p^k q_q^k$ valid when $pq > M$ and $\pi_p + \pi_q \leq 1/3$, with an explicit constant $C_0(c)$ depending only on $c$, together with an $L^2$ Minkowski summation that controls the covariances among all distinct prime-pair terms.
\end{abstract}

\noindent\textbf{Keywords:} Random LCM, Euler totient, coprime density, central limit theorem, martingale central limit theorem, Doob decomposition, Hall--Heyde, probabilistic number theory.

\noindent\textbf{MSC 2020:} 60F05, 11N37, 11K65, 60G42.

\section{Introduction}\label{sec:intro}

The arithmetic structure of the least common multiple of a random set of integers has been studied extensively in the regime where the support size $n$ grows while the number of samples $k$ stays fixed or bounded; see Cilleruelo, Ru\'e, \v{S}arka, and Zumalac\'arregui~\cite{CRSZ}; Hilberdink and T\'oth~\cite{HT}; Alsmeyer, Kabluchko, and Marynych~\cite{AKM}; Bostan, Marynych, and Raschel~\cite{BMR}; Buraczewski, Iksanov, and Marynych~\cite{BIM}; Iksanov, Marynych, and Raschel~\cite{IMR}; Kabluchko, Marynych, and Raschel~\cite{KMR}; and Sanna~\cite{Sanna}. These works establish limit theorems for $\log \lcm$ and for multiplicative functions of the LCM in the fixed-$k$, $n \to \infty$ regime.

A complementary regime, in which the support $\{2, \ldots, M\}$ is fixed and the sample size $k \to \infty$, was introduced in the companion paper~\cite{wakid1}. There the natural object of study is not the LCM itself but the coprime density
\[
\frac{\varphi(L_k)}{L_k} = \prod_{p \mid L_k}\!\left(1 - \frac{1}{p}\right),
\]
whose large-$k$ behaviour is governed by exactly which primes $p \leq M$ appear as divisors of $L_k = \lcm(X_1, \ldots, X_k)$. The paper~\cite{wakid1} established, writing $\rho_k(M) := \EE[\varphi(L_k)/L_k]$, $P_M := \prod_{p \leq M}(1 - 1/p)$, $m_p := \lfloor M/p \rfloor$, $\pi_p := m_p/(M-1)$, $q_p := 1 - \pi_p$:
\begin{itemize}[leftmargin=2em]\itemsep=2pt
\item \emph{Exact identity:}
\[
\rho_k(M)=P_M\sum_{T\subseteq\PM}\beta_T(M)^k
\prod_{p\in T}(p-1)^{-1},
\]
where $\beta_T(M)=\PP(\gcdop(X,\prod_{p\in T}p)=1)$ for a single uniform draw $X$.
\item \emph{Fixed-$M$ limit with sharp rate:} $\rho_k(M) \to P_M$ as $k \to \infty$, with an explicit geometric rate of convergence controlled by $(q^{**}(M)/q^*(M))^k$.
\item \emph{Joint-regime first-order asymptotic:} in the regime $k = \lfloor cM \rfloor$, $M \to \infty$, fixed $c > 0$,
\[
\ln M \cdot \frac{\rho_k(M) - P_M}{P_M} \longrightarrow A(c) := \sum_{j \geq 1} e^{-cj}\ln(1 + 1/j).
\]
\end{itemize}

\paragraph{Dependence on the companion paper.}
The proofs below are self-contained apart from three explicitly identified inputs from~\cite{wakid1}: the subset-density bounds collected in Lemma~\ref{lem:beta-basic}, the level-count estimate quoted in Lemma~\ref{lem:envelope}, and the elementary disjointness observation quoted in Proposition~\ref{prop:disjoint}. The exact identity and first-order asymptotic for $\rho_k(M)$ are recalled only to situate the present second-order results and are not used in the proofs of Theorems~\ref{thm:variance} and~\ref{thm:CLT}. The disjointness identity needed later is also rederived directly in Lemma~\ref{lem:beta-pq-exact}.

In this paper we study the second-order behaviour of the centred log coprime density
\[
W_k(M) := \ln(\varphi(L_k)/L_k) - \EE\,\ln(\varphi(L_k)/L_k),
\]
in the same joint regime $k = \lfloor cM \rfloor$, $M \to \infty$, with fixed $c > 0$. We prove two results.

\begin{theorem}[Variance asymptotic]\label{thm:variance}
For each fixed $c > 0$, in the joint regime $k = \lfloor cM \rfloor$, $M \to \infty$,
\begin{equation}\label{eq:variance-limit}
M \ln M \cdot s_k^2 \;\longrightarrow\; B(c) := \sum_{j \geq 1} e^{-cj}(1 - e^{-cj}) = \frac{1}{e^c - 1} - \frac{1}{e^{2c} - 1},
\end{equation}
where $s_k^2 = \Var W_k(M)$.
\end{theorem}

\begin{theorem}[Central limit theorem]\label{thm:CLT}
For each fixed $c > 0$, in the joint regime $k = \lfloor cM \rfloor$, $M \to \infty$,
\[
\frac{W_k(M)}{s_k} \;\xrightarrow{d}\; \NN(0, 1).
\]
Consequently,
\[
\sqrt{M \ln M} \cdot \frac{W_k(M)}{\sqrt{B(c)}} \;\xrightarrow{d}\; \NN(0, 1),
\qquad
\sqrt{M \ln M} \cdot \frac{\varphi(L_k)/L_k - \rho_k(M)}{P_M \sqrt{B(c)}} \;\xrightarrow{d}\; \NN(0, 1).
\]
The first consequence follows from Theorem~\ref{thm:variance} and Slutsky's theorem; the second follows from the linear-scale transfer Lemma~\ref{lem:linear-transfer} below.
\end{theorem}

\subsection{Method and outline}

Theorem~\ref{thm:variance} is proved by a direct diagonal / off-diagonal decomposition. Writing $X_p = \one\{p \mid L_k\}$ so that $\ln(\varphi(L_k)/L_k) = \sum_{p \leq M} X_p \ln(1 - 1/p)$, the variance decomposes as
\[
s_k^2 = V_1(k, M) + V_2(k, M)
\]
with $V_1$ the diagonal variance contribution and $V_2$ the off-diagonal covariance sum. The diagonal $V_1$ is handled by partitioning primes by level $m_p = j$ and applying a partial-summation analogue of Mertens' second theorem that yields $\sum_{p \colon m_p = j} 1/p^2 \sim 1/(M \ln M)$ for each fixed $j$, giving $M \ln M \cdot V_1 \to B(c)$ by dominated convergence. The off-diagonal $V_2$ is bounded using the exact covariance identity
\[
\beta_{\{p,q\}}(M) - q_p q_q = -\frac{m_p m_q}{(M-1)^2} \quad \text{for } pq > M,
\]
together with a mean-value bound. Section~\ref{sec:variance-proof} carries this out in detail.

Theorem~\ref{thm:CLT} is proved via the Hall--Heyde martingale central limit theorem~\cite[Thm.~3.2]{HH}, applied to the Doob filtration $\F_i = \sigma(X_1, \ldots, X_i)$. Section~\ref{sec:martingale} derives the martingale decomposition $W_k = \sum_{i=1}^k D_i$ and identifies the explicit form of the increments.

The application of~\cite{HH} requires verifying two conditions. Section~\ref{sec:lindeberg} proves the conditional Lindeberg condition via a uniform bound on the increments: $\max_i |D_i| = O((\log M)^3/M) = o(s_k)$ on a super-polynomially likely event. The bound exploits two facts simultaneously. First, a coupon-collector tail estimate: with super-polynomially high probability, every prime $p$ with $m_p \geq (\log M)^2$ has been sampled by time $k/2$. Second, a sieve-geometric observation: once contributing primes are restricted to $p > M/(\log M)^2$, at most one of them can divide any given $x \leq M$, because $(M/(\log M)^2)^2 > M$ for $M$ sufficiently large. Either fact alone gives only a weak bound; together they yield the required estimate.

Section~\ref{sec:condvar} proves the conditional-variance concentration $\sum_{i=1}^k \EE[D_i^2 \mid \F_{i-1}]/s_k^2 \xrightarrow{P} 1$. The argument rests on two explicit moment bounds. Lemma~\ref{lem:bd1} controls the per-prime variance $\Var(R_{pp}^{(k)})$ via a substitution $Y_p = q_p^{-2(T_p \wedge k)}$ that reduces the quantity to a function of a single geometric random variable. Lemma~\ref{lem:bd2} bounds the cross-prime covariance $|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})|$ for $pq > M$ via a comparison between the true joint law of the hitting times $(T_p, T_q)$ and its independent product. These bounds are combined with a single-pair second-moment estimate, a level-summation argument, and Minkowski's inequality in $L^2$, which controls all cross-covariances among the prime-pair summands without assuming orthogonality.

Section~\ref{sec:conclusion-clt} assembles the pieces for Theorem~\ref{thm:CLT}. Section~\ref{sec:numerics} presents numerical diagnostics for both theorems.

\section{Preliminaries}\label{sec:prelim}

Throughout, $M \geq 2$ is a fixed integer, $X_1, X_2, \ldots$ are i.i.d.\ uniform on $\{2, \ldots, M\}$, $L_k = \lcm(X_1, \ldots, X_k)$, and $\rho_k(M) = \EE[\varphi(L_k)/L_k]$. For a prime $p \leq M$, set
\[
m_p := \lfloor M/p \rfloor, \quad \pi_p := \frac{m_p}{M-1}, \quad q_p := 1 - \pi_p, \quad \xi_p^{(i)} := \one\{p \mid X_i\} - \pi_p.
\]
Then $m_p$ is the number of multiples of $p$ in $\{2, \ldots, M\}$ and $\pi_p = \PP(p \mid X_i)$ for any single draw. For $T \subseteq \PM := \{p : p \leq M\}$, let $A_T(M) = \{n \in \{2, \ldots, M\} : \gcdop(n, \prod_{p \in T} p) = 1\}$, so $\beta_T(M) = |A_T(M)|/(M-1)$ is the probability that a single uniform draw is coprime to every prime in $T$; in particular $\beta_\emptyset = 1$ and $\beta_{\{p\}} = q_p$. Let $T_p := \min\{i \geq 1 : p \mid X_i\}$, which is geometric with success probability $\pi_p$. Let $\F_i := \sigma(X_1, \ldots, X_i)$ and $\F_0 := \{\varnothing, \Omega\}$. Logarithms are natural throughout, and $c > 0$ is fixed with $k := \lfloor cM \rfloor$ unless otherwise stated. We write $C$ for an absolute constant and $C(c)$ for a constant depending only on $c$; these may vary from line to line.

We record three elementary facts and the basic subset-density lemma.

\begin{lemma}[Elementary subset-density bounds; {\cite[Lems.~2.1--2.4]{wakid1}}]\label{lem:beta-basic}
For every $T \subseteq \PM$:
\begin{enumerate}[label=(\roman*), leftmargin=2em]
\item $\PP(\text{no prime in $T$ divides $L_k$}) = \beta_T(M)^k$;
\item $\beta_T(M) \leq q_p(M)$ for every $p \in T$, $T \neq \emptyset$;
\item if $|T| = r \geq 1$, then $\beta_T(M)^k \leq \prod_{p \in T} q_p(M)^{k/r}$ for every real $k > 0$.
\end{enumerate}
\end{lemma}

\begin{lemma}[Rosser--Schoenfeld level envelope; {\cite{RosserSchoenfeld1962}, \cite[Lem.~6.3]{wakid1}}]\label{lem:envelope}
For all $M \geq 289$ and all $1 \leq j \leq M/17$,
\[
\bigl|\{p \leq M : m_p = j\}\bigr| \leq \frac{1.26\,(M/j)}{\ln(M/j)}, \qquad \sum_{p \colon m_p = j} \frac{1}{p-1} \leq \frac{5.04}{\ln(M/j)}.
\]
In particular, for $1 \leq j \leq M^{1/2}$,
\[
\bigl|\{p \leq M : m_p = j\}\bigr| \leq \frac{2.52\,M}{j\ln M},
\qquad
\sum_{p \colon m_p = j} \frac{1}{p-1} \leq \frac{10.08}{\ln M}.
\]
\end{lemma}

\begin{proposition}[Disjointness for $pq > M$; {\cite[Rem.~5.6]{wakid1}}]\label{prop:disjoint}
For distinct primes $p, q \leq M$ with $pq > M$, the events $\{p \mid X\}$ and $\{q \mid X\}$ are disjoint under a single uniform draw $X$ on $\{2, \ldots, M\}$; equivalently $\PP(p \mid X, q \mid X) = 0$, so $\PP(p \nmid X, q \nmid X) = 1 - \pi_p - \pi_q$ and
\[
\beta_{\{p,q\}}(M) - q_p(M)\, q_q(M) = -\pi_p \pi_q.
\]
\end{proposition}

Part (iii) of Lemma~\ref{lem:beta-basic} is the geometric-mean bound of~\cite[Lem.~2.4]{wakid1}, which underlies the sharp joint-scaling result in the companion paper.

\section{Variance asymptotic: proof of Theorem~\ref{thm:variance}}\label{sec:variance-proof}

Let $k = \lfloor cM \rfloor$. Write $\varphi(L_k)/L_k = \prod_{p \leq M}(1 - 1/p)^{X_p}$, where $X_p = \one\{p \mid L_k\}$, so
\[
\ln\bigl(\varphi(L_k)/L_k\bigr) = \sum_{p \leq M} X_p \ln(1 - 1/p).
\]
The variance of $X_p$ is $(1 - q_p^k) q_p^k$ and for $p \neq q$,
\[
\Cov(X_p, X_q) = \PP(p \mid L_k,\; q \mid L_k) - (1 - q_p^k)(1 - q_q^k) = \beta_{\{p,q\}}^k - q_p^k q_q^k,
\]
as follows from the inclusion-exclusion identity $\PP(p \mid L_k, q \mid L_k) = 1 - q_p^k - q_q^k + \beta_{\{p,q\}}^k$. Therefore
\begin{equation}\label{eq:Var-W-decomp}
s_k^2 = V_1(k, M) + V_2(k, M),
\end{equation}
where the diagonal and off-diagonal contributions are
\begin{align*}
V_1(k, M) &= \sum_{p \leq M} \ln(1 - 1/p)^2\, q_p^k (1 - q_p^k),\\
V_2(k, M) &= \sum_{p \neq q,\; p, q \leq M} \ln(1 - 1/p) \ln(1 - 1/q)\bigl(\beta_{\{p,q\}}^k - q_p^k q_q^k\bigr).
\end{align*}

\subsection{Diagonal contribution}\label{sec:variance-V1}

We show $M \ln M \cdot V_1(k, M) \to B(c)$. Partition by $m_p = j$:
\[
V_1(k, M) = \sum_{j \geq 1}\bigl(1 - j/(M-1)\bigr)^k\bigl(1 - (1 - j/(M-1))^k\bigr) \sum_{p \colon m_p = j} \ln(1 - 1/p)^2.
\]
For $p \in (M/(j+1), M/j]$, $\ln(1 - 1/p)^2 = 1/p^2 + O(1/p^3)$, and since $p \geq M/(j+1)$ we have $1/p^3 = O((j+1)^3/M^3)$. Hence
\[
\sum_{p \colon m_p = j} \ln(1 - 1/p)^2 = \sum_{p \colon m_p = j} \frac{1}{p^2} + O\!\left(\frac{(j+1)^3}{M^3}\cdot \pi(M/j)\right) = \sum_{p \colon m_p = j} \frac{1}{p^2} + O\!\left(\frac{(j+1)^2}{M^2 \ln M}\right).
\]

\begin{lemma}[Level-$j$ sum of reciprocal squares]\label{lem:1-over-p-squared}
For each fixed $j \geq 1$,
\[
M \ln M \cdot \sum_{p \colon m_p = j} \frac{1}{p^2} \longrightarrow 1 \qquad (M \to \infty).
\]
\end{lemma}
\begin{proof}
By partial summation and the prime number theorem in its standard quantitative form~\cite{Tenenbaum2015}, $\sum_{p \leq x} 1/p^2 = \pi(x)/x^2 + 2 \int_2^x \pi(t)/t^3\, dt$, so $\sum_{p \in (A, B]} 1/p^2 = \pi(B)/B^2 - \pi(A)/A^2 + 2\int_A^B \pi(t)/t^3\, dt$. With $A = M/(j+1)$, $B = M/j$, and using $\pi(t) = t/\ln t + O(t/(\ln t)^2)$,
\[
\pi(B)/B^2 = \frac{1}{B \ln B}(1 + O(1/\ln B)) = \frac{j}{M \ln(M/j)}(1 + o(1)) = \frac{j}{M \ln M}(1 + o(1)),
\]
and analogously $\pi(A)/A^2 = (j+1)/(M \ln M)(1+o(1))$. Their difference is
\[
\pi(B)/B^2 - \pi(A)/A^2 = -\frac{1}{M \ln M}(1+o(1)).
\]
The integral is
\[
2\int_A^B \frac{1}{t^2 \ln t}(1 + O(1/\ln t))\, dt = \frac{2}{\ln M}\int_A^B \frac{dt}{t^2}(1 + o(1)) = \frac{2}{\ln M}\!\left(\frac{1}{A} - \frac{1}{B}\right)(1 + o(1)),
\]
and $1/A - 1/B = (j+1)/M - j/M = 1/M$, so the integral contributes $2/(M\ln M)(1+o(1))$. Summing the two contributions:
\[
\sum_{p \colon m_p = j} \frac{1}{p^2} = \left[-\frac{1}{M \ln M} + \frac{2}{M \ln M}\right](1+o(1)) = \frac{1}{M \ln M}(1 + o(1)),
\]
for each fixed $j \geq 1$.
\end{proof}

\emph{Dominated convergence.} For each fixed $j$, Lemma~\ref{lem:1-over-p-squared} together with $(1 - j/(M-1))^k \to e^{-cj}$ and $(1 - (1 - j/(M-1))^k) \to 1 - e^{-cj}$ gives
\[
M \ln M \cdot (1 - j/(M-1))^k\bigl(1 - (1 - j/(M-1))^k\bigr) \sum_{p \colon m_p = j} \ln(1-1/p)^2 \longrightarrow e^{-cj}(1 - e^{-cj}).
\]
A uniform envelope follows directly from the prime-count part of Lemma~\ref{lem:envelope}, applied to the summand $\ln(1-1/p)^2$ itself rather than to its leading term $1/p^2$. If $m_p=j$, then $p>M/(j+1)$, so $p-1 > M/(2(j+1))$ whenever $M \geq 2(j+1)$, and hence $\ln(1-1/p)^2 \leq (p-1)^{-2} < 4(j+1)^2/M^2$. Therefore, for $1 \leq j \leq M^{1/2}$ and $M \geq 289$ (which guarantees both $\ln(M/j) \geq \tfrac12 \ln M$ and $M \geq 2(j+1)$ in this range),
\[
\sum_{p \colon m_p = j} \ln(1-1/p)^2
\leq \bigl|\{p:m_p=j\}\bigr|\frac{4(j+1)^2}{M^2}
\leq \frac{2.52\,M}{j\ln M}\cdot\frac{4(j+1)^2}{M^2}
=\frac{10.08\,(j+1)^2}{Mj\ln M}.
\]
For the exponential factor, since $k \geq cM - 1$ and $1 - x \leq e^{-x}$,
\[
(1 - j/(M-1))^k \leq e^{-kj/(M-1)} \leq e^{-cj}\, e^{j/(M-1)} \leq 2 e^{-cj}
\qquad (1 \leq j \leq M^{1/2},\ M \geq 5).
\]
Hence each level-$j$ term of $M \ln M \cdot V_1$ is bounded by $10.08\,(j+1)^2/j \cdot 2 e^{-cj}$, which is summable in $j$. The tail $j > M^{1/2}$ decays super-exponentially, since there $(1 - j/(M-1))^k \leq e^{-c\sqrt{M}/2}$ for $M \geq M_0(c)$ while $\sum_{p \leq M} \ln(1-1/p)^2 \leq \sum_{n \geq 2}(n-1)^{-2} = O(1)$. By dominated convergence,
\begin{equation}\label{eq:V1-limit}
\lim_{M \to \infty} M \ln M \cdot V_1(k, M) = \sum_{j \geq 1} e^{-cj}(1 - e^{-cj}) = B(c).
\end{equation}

\subsection{Off-diagonal contribution is negligible}\label{sec:variance-V2}

We show $M \ln M \cdot V_2(k, M) \to 0$. The starting point is an exact identity valid in a restricted range of prime pairs.

\begin{lemma}[Covariance identity for disjoint prime levels]\label{lem:beta-pq-exact}
Let $p, q \in \PM$ with $p \neq q$ and $pq > M$. Then
\begin{equation}\label{eq:beta-pq-exact}
\beta_{\{p,q\}}(M) - q_p(M)\, q_q(M) = -\frac{m_p\, m_q}{(M-1)^2}.
\end{equation}
\end{lemma}
\begin{proof}
By inclusion-exclusion, $|A_{\{p,q\}}(M)| = M - \lfloor M/p \rfloor - \lfloor M/q \rfloor + \lfloor M/(pq) \rfloor - 1$. When $pq > M$, $\lfloor M/(pq) \rfloor = 0$, so $(M-1)\beta_{\{p,q\}} = M - 1 - m_p - m_q$. Hence
\[
(M-1)^2\bigl(\beta_{\{p,q\}} - q_p q_q\bigr) = (M-1)(M - 1 - m_p - m_q) - (M-1-m_p)(M-1-m_q) = -m_p m_q. \qedhere
\]
\end{proof}

\begin{remark}\label{rem:pq-gt-M-automatic}
For levels $(j_1, j_2)$ with $(j_1+1)(j_2+1) < M$, every prime $p$ with $m_p = j_1$ and $q$ with $m_q = j_2$ satisfies $pq \geq M/(j_1+1) \cdot M/(j_2+1) = M^2/((j_1+1)(j_2+1)) > M$, so~\eqref{eq:beta-pq-exact} applies. In particular, if $\max(j_1, j_2) \leq M^{1/4}$ then $(j_1+1)(j_2+1) \leq (M^{1/4}+1)^2 < M$ for all $M \geq 7$, so the identity applies to every pair in this range.
\end{remark}

\emph{Main-range bound.}
Let $\mathcal{J}(M) = \{(p,q) : p \neq q,\, p, q \leq M,\, \max(m_p, m_q) \leq M^{1/4}\}$ and write
\[
V_2^{\mathrm{main}}(k, M) := \sum_{(p,q) \in \mathcal{J}(M)} \ln(1 - 1/p) \ln(1 - 1/q) \bigl(\beta_{\{p,q\}}^k - q_p^k q_q^k\bigr),
\]
with $V_2^{\mathrm{tail}}(k, M) := V_2(k, M) - V_2^{\mathrm{main}}(k, M)$.

For $(p,q) \in \mathcal{J}(M)$, let $j_1 = m_p$, $j_2 = m_q$. By Remark~\ref{rem:pq-gt-M-automatic} and Lemma~\ref{lem:beta-pq-exact}, $\beta_{\{p,q\}} - q_p q_q = -j_1 j_2/(M-1)^2$; in particular $\beta_{\{p,q\}} < q_p q_q$, so by the mean-value form $|a^k - b^k| \leq k|a-b|\max(a,b)^{k-1}$ with $a = \beta_{\{p,q\}}, b = q_p q_q$,
\begin{equation}\label{eq:Delta-bound}
\bigl|\beta_{\{p,q\}}^k - q_p^k q_q^k\bigr| \leq \frac{k\, j_1 j_2}{(M-1)^2}\,(q_p q_q)^{k-1}.
\end{equation}
Since $q_p q_q \geq 1 - (j_1 + j_2)/(M-1) \geq 1/2$ for $M$ large enough (because $j_1+j_2 \leq 2 M^{1/4} = o(M)$), we have $(q_p q_q)^{k-1} \leq 2 (q_p q_q)^k$. Using also $|\ln(1 - 1/p)| \leq 1/(p-1)$, the contribution of any pair $(p,q) \in \mathcal{J}(M)$ with $m_p = j_1, m_q = j_2$ is at most
\[
\frac{1}{(p-1)(q-1)} \cdot \frac{2k\, j_1 j_2}{(M-1)^2}\,(q_p q_q)^k.
\]
Summing over pairs at fixed levels $(j_1, j_2)$ --- noting $(q_p q_q)^k$ depends on the pair only through $j_1, j_2$ --- and using $S_j(M) := \sum_{p \colon m_p = j} 1/(p-1)$,
\[
\sum_{\substack{(p,q) \in \mathcal{J}(M) \\ m_p = j_1,\, m_q = j_2}} \frac{1}{(p-1)(q-1)} \leq S_{j_1}(M)\, S_{j_2}(M).
\]
By Lemma~\ref{lem:envelope}, $S_j(M) \leq 10.08/\ln M$ uniformly for $1 \leq j \leq M^{1/2}$ and $M \geq 289$, hence certainly for $j \leq M^{1/4}$. Moreover, $(q_p q_q)^k \leq 2\, e^{-c(j_1+j_2)}$ uniformly over $j_1, j_2 \leq M^{1/4}$ once $M \geq M_0(c)$: indeed $\ln(1-x) = -x + O(x^2)$ and $k/(M-1) = c + O(1/M)$ give
\begin{equation}\label{eq:uniform-taylor}
k \ln(q_p q_q) = -\frac{k(j_1+j_2)}{M-1} + O\!\left(\frac{k(j_1^2+j_2^2)}{M^2}\right) = -c(j_1+j_2) + O\bigl(M^{-1/2}\bigr),
\end{equation}
uniformly in this range, since $j_1^2 + j_2^2 \leq 2M^{1/2}$. Therefore
\[
|V_2^{\mathrm{main}}(k, M)| \leq \frac{4k}{(M-1)^2}\cdot \frac{(10.08)^2}{(\ln M)^2}\cdot \sum_{j_1, j_2 \geq 1} j_1 j_2\, e^{-c(j_1 + j_2)}.
\]
The double sum factorises:
\[
\sum_{j_1, j_2 \geq 1} j_1 j_2\, e^{-c(j_1+j_2)} = \left(\sum_{j \geq 1} j\, e^{-cj}\right)^{\!2} = \frac{e^{2c}}{(e^c - 1)^4}.
\]
With $k = \lfloor cM \rfloor$ so $k/(M-1)^2 = O(1/M)$,
\begin{equation}\label{eq:V2-main-bound}
|V_2^{\mathrm{main}}(k, M)| = O\!\left(\frac{1}{M\,(\ln M)^2}\right).
\end{equation}

\emph{Tail-range bound.}
For $(p,q) \notin \mathcal{J}(M)$, at least one of $m_p, m_q$ exceeds $M^{1/4}$. Say $m_p > M^{1/4}$; then $q_p^k \leq (1 - M^{1/4}/(M-1))^{\lfloor cM \rfloor} \leq \exp(-c M^{1/4}(1+o(1)))$. Using Lemma~\ref{lem:beta-basic}(ii) in the form $\beta_{\{p,q\}} \leq q_p$, $\beta_{\{p,q\}}^k \leq q_p^k$, and the trivial bound $q_p^k q_q^k \leq q_p^k$,
\[
\bigl|\beta_{\{p,q\}}^k - q_p^k q_q^k\bigr| \leq 2\, q_p^k.
\]
Writing $\Lambda(M) := \sum_{p \leq M} 1/(p-1) = O(\ln \ln M)$ (Mertens),
\begin{align*}
|V_2^{\mathrm{tail}}(k, M)| &\leq 2 \sum_{p \colon m_p > M^{1/4}} \frac{q_p^k}{p-1} \sum_{q \leq M,\, q \neq p} \frac{1}{q-1} + (\text{symmetric term with }p, q\text{ swapped})\\
&\leq 4\, \Lambda(M)\, \exp(-c M^{1/4}(1+o(1)))\, \Lambda(M)\\
&= O\bigl((\ln \ln M)^2\, e^{-c M^{1/4}}\bigr),
\end{align*}
which vanishes super-polynomially in $M$. In particular $M \ln M \cdot V_2^{\mathrm{tail}} \to 0$.

Combining~\eqref{eq:V2-main-bound} and the tail bound,
\begin{equation}\label{eq:V2-total-bound}
|V_2(k, M)| = O\!\left(\frac{1}{M\,(\ln M)^2}\right),
\end{equation}
so $M \ln M \cdot |V_2(k, M)| = O(1/\ln M) \to 0$. Combined with~\eqref{eq:V1-limit}, this gives~\eqref{eq:variance-limit}. \qed

\begin{remark}[Sign of $V_2$]\label{rem:V2-sign}
The leading contribution to $V_2$ comes from the region $\max(m_p, m_q) \leq M^{1/4}$. There $\beta_{\{p,q\}} - q_p q_q = -m_p m_q/(M-1)^2 < 0$, so $\beta_{\{p,q\}}^k - q_p^k q_q^k < 0$. The weights $\ln(1-1/p)\ln(1-1/q)$ are positive. Hence the main-region sum, and therefore $V_2$ for $M$ large, is strictly negative. Numerically (Section~\ref{sec:numerics}) $V_2$ is observed negative at every tested $(k, M)$.
\end{remark}

\begin{remark}[$s_k \asymp (M \ln M)^{-1/2}$]\label{rem:sk-asymp}
Theorem~\ref{thm:variance} gives $s_k^2 \sim B(c)/(M \ln M)$ with $B(c) > 0$ for every $c > 0$, so in particular $s_k \asymp (M \ln M)^{-1/2}$ in the joint regime. This scale is used throughout the CLT proof below.
\end{remark}

\section{Martingale decomposition}\label{sec:martingale}

Let $F_k := \ln(\varphi(L_k)/L_k) = \sum_{p \leq M} \ln(1 - 1/p)\, \one\{p \mid L_k\}$ and $D_i := \EE[F_k \mid \F_i] - \EE[F_k \mid \F_{i-1}]$ for $1 \leq i \leq k$.

\begin{theorem}[Doob decomposition]\label{thm:doob}
$(D_i, \F_i)_{i=1}^k$ is a martingale-difference sequence with $\sum_{i=1}^k D_i = W_k$, and
\begin{equation}\label{eq:Di}
D_i = \sum_{\substack{p \leq M \\ p \nmid L_{i-1}}} \ln(1 - 1/p)\, q_p^{k-i}\, \xi_p^{(i)}.
\end{equation}
\end{theorem}

\begin{proof}
By linearity, $\EE[F_k \mid \F_i] = \sum_p \ln(1 - 1/p)\, \EE[\one\{p \mid L_k\} \mid \F_i]$. Since $p \mid L_k$ iff $p \mid L_i$ or $p \mid X_j$ for some $j > i$, and $X_{i+1}, \ldots, X_k \perp \F_i$,
\[
\EE[\one\{p \mid L_k\} \mid \F_i] = \one\{p \mid L_i\} + \one\{p \nmid L_i\}\bigl(1 - q_p^{k-i}\bigr).
\]
On $\{p \mid L_{i-1}\}$, both $\EE[\one\{p \mid L_k\} \mid \F_i]$ and $\EE[\one\{p \mid L_k\} \mid \F_{i-1}]$ equal $1$, so this prime contributes nothing to $D_i$. On $\{p \nmid L_{i-1}\}$, case analysis on $\one\{p \mid X_i\} \in \{0, 1\}$ gives
\[
\EE[\one\{p \mid L_k\} \mid \F_i] - \EE[\one\{p \mid L_k\} \mid \F_{i-1}] = \begin{cases} q_p^{k-i+1}, & p \mid X_i, \\ -q_p^{k-i}(1 - q_p), & p \nmid X_i, \end{cases}
\]
which equals $q_p^{k-i}\, \xi_p^{(i)}$ in either case. Summing over $p$ gives~\eqref{eq:Di}; martingale-difference and telescoping properties are immediate.
\end{proof}

By orthogonality of martingale increments and Theorem~\ref{thm:variance}, $\sum_{i=1}^k \EE[D_i^2] = s_k^2 \sim B(c)/(M \ln M)$.

\section{Conditional Lindeberg condition}\label{sec:lindeberg}

Let $i_0 := \lceil k/2\rceil$ and define the event
\[
\Omega^* := \bigcap_{\substack{p \leq M \\ m_p \geq (\log M)^2}} \{p \mid L_{i_0-1}\}.
\]

\begin{theorem}[Uniform increment bound]\label{thm:supnorm}
For every fixed $c > 0$ and every $K \geq 0$, $\PP((\Omega^*)^c) = o(M^{-K})$ as $M \to \infty$. Moreover, there exists $M_0 = M_0(c)$ such that, for $M \geq M_0$,
\begin{equation}\label{eq:supnorm}
\max_{1 \leq i \leq k}|D_i| \leq C(c)\,(\log M)^3/M = o(s_k)
\end{equation}
on $\Omega^*$. More precisely, for $i < i_0$ the bound holds almost surely, while for $i \geq i_0$ it holds uniformly over the possible value of $X_i$ whenever the history belongs to $\Omega^*$.
\end{theorem}

\begin{proof}
\textit{Probability of $\Omega^*$.} For a prime $p$ with $m_p \geq (\log M)^2$,
\[
\PP(p \nmid L_{i_0-1}) = q_p^{i_0-1} \leq \exp\bigl(-\pi_p(i_0-1)\bigr) \leq \exp\bigl(-(c/2)(\log M)^2(1+o(1))\bigr).
\]
A union bound over at most $\pi(M) \leq M/\ln M$ primes gives
\[
\PP((\Omega^*)^c) \leq \frac{M}{\ln M}\exp\bigl(-(c/2)(\log M)^2(1+o(1))\bigr)=o(M^{-K})
\]
for every $K$.

\textit{Case A: $i \geq i_0$, on $\Omega^*$.}
Decompose $D_i = S_i(X_i) - C_i$, where
\[
S_i(x) := \sum_{\substack{p \mid x \\ p \nmid L_{i-1}}} \ln(1 - 1/p)\, q_p^{k-i}, \qquad C_i := \sum_{p \nmid L_{i-1}} \ln(1 - 1/p)\, q_p^{k-i}\, \pi_p.
\]
On $\Omega^*$, every contributing prime satisfies $m_p < (\log M)^2$, hence $p > M/(\log M)^2$. For $M \geq M_0(c)$ large enough that $M > (\log M)^4$, any two such primes $p_1, p_2$ obey
\[
p_1 p_2 > \bigl(M/(\log M)^2\bigr)^2 = M^2/(\log M)^4 > M,
\]
so $p_1$ and $p_2$ cannot both divide any $x \in \{2, \ldots, M\}$. Hence at most one contributing prime divides $x$, giving
\[
|S_i(x)| \leq \max_{\substack{p \nmid L_{i-1} \\ m_p < (\log M)^2}} \frac{q_p^{k-i}}{p-1} \leq \frac{(\log M)^2}{M-1}.
\]

For $C_i$, a prime at level $j \leq (\log M)^2$ contributes at most $|\ln(1-1/p)| \pi_p \leq 1/(p-1) \cdot j/(M-1) \leq 2(j+1)/M \cdot (j+1)/M \leq 2(j+1)^2/M^2$ (using $q_p^{k-i} \leq 1$, the crudest factor since $k-i$ may be as small as zero when $i = k$). Summing using $|\{p : m_p = j\}| \leq 1.26\, M/(j \ln(M/j))$ of Lemma~\ref{lem:envelope}:
\[
|C_i| \leq \sum_{j=1}^{(\log M)^2} \frac{1.26\, M}{j \ln(M/j)} \cdot \frac{2(j+1)^2}{M^2} \leq \frac{5.04}{M \ln M} \sum_{j=1}^{(\log M)^2} \frac{(j+1)^2}{j} = O\!\left(\frac{(\log M)^3}{M}\right),
\]
since $\sum_{j=1}^{J}(j+1)^2/j = \sum_j (j + 2 + 1/j) = O(J^2)$ at $J = (\log M)^2$.

Combining, $|D_i| \leq |S_i(X_i)|+|C_i| \leq C(c)(\log M)^3/M$ uniformly over $X_i$ on $\Omega^*$.

\textit{Case B: $i < i_0$, unconditionally.}
Put $r:=k-i$. Since $i<\lceil k/2\rceil$, one has $r>k/2$; hence, for all sufficiently large $M$, $r\geq cM/3$. We again write $D_i=S_i(X_i)-C_i$.

For the part of $S_i(x)$ coming from primes $p\geq\sqrt M$, at most one such prime can divide a fixed $x\leq M$: two distinct prime divisors in this range would have product $>M$. If that prime has level $m_p=j\leq\sqrt M$, then
\[
\frac{q_p^r}{p-1}
\leq \frac{2(j+1)}{M}
\exp\!\left(-\frac{rj}{M-1}\right)
\leq \frac{2(j+1)}{M}e^{-cj/4}
=O_c(M^{-1}),
\]
uniformly in $j$. For primes $p<\sqrt M$, one has $m_p\geq\sqrt M-1$, so
\[
q_p^r\leq \exp\!\left(-\frac{r m_p}{M-1}\right)
\leq e^{-c\sqrt M/4}
\]
for large $M$. Consequently their total contribution is at most
$e^{-c\sqrt M/4}\sum_{p\leq M}(p-1)^{-1}=O(e^{-c\sqrt M/4}\log\log M)$.
Thus $\sup_x|S_i(x)|=O_c(M^{-1})$.

For the centring term, split by levels. When $j\leq\sqrt M$, Lemma~\ref{lem:envelope}, $q_p^r\leq e^{-cj/4}$, $\pi_p=j/(M-1)$, and $(p-1)^{-1}\leq2(j+1)/M$ give
\begin{align*}
\sum_{\substack{p:m_p=j}} |\ln(1-1/p)|\,\pi_p q_p^r
&\leq \frac{C}{M\ln M}(j+1)e^{-cj/4}.
\end{align*}
Summing over $j$ yields $O_c((M\ln M)^{-1})$. The levels $j>\sqrt M$ contribute
$O(e^{-c\sqrt M/4}\log\log M)$. Therefore $|C_i|=O_c((M\ln M)^{-1})$ and
$|D_i|=O_c(M^{-1})$ uniformly over all histories and all values of $X_i$.

\textit{Comparison to $s_k$.}
$(\log M)^3/M \cdot (M \ln M)^{1/2} = (\log M)^{7/2}/\sqrt{M} \to 0$, establishing~\eqref{eq:supnorm}.
\end{proof}

\begin{corollary}[Conditional Lindeberg]\label{cor:HH2}
For every $\varepsilon > 0$,
\[
\frac{1}{s_k^2} \sum_{i=1}^k \EE\!\left[D_i^2 \one\{|D_i| > \varepsilon s_k\} \,\middle|\, \F_{i-1}\right] \xrightarrow{P} 0.
\]
\end{corollary}

\begin{proof}
By Remark~\ref{rem:sk-asymp}, $s_k \asymp (M\ln M)^{-1/2}$, and hence $(\log M)^3/M=o(s_k)$. For $M$ sufficiently large, Theorem~\ref{thm:supnorm} implies that every summand in the conditional Lindeberg expression vanishes on $\Omega^*$: for $i<i_0$ this holds unconditionally, and for $i\geq i_0$ it holds uniformly over $X_i$ conditional on any history in $\Omega^*$. Therefore the entire nonnegative conditional Lindeberg sum is zero on $\Omega^*$. Consequently, for every $\delta>0$,
\[
\PP\!\left(\frac{1}{s_k^2}\sum_{i=1}^k \EE\!\left[D_i^2\one\{|D_i|>\varepsilon s_k\}\mid\F_{i-1}\right]>\delta\right)
\leq \PP((\Omega^*)^c)\longrightarrow 0.
\]
\end{proof}

\section{Conditional-variance concentration}\label{sec:condvar}

Let $U_i := \EE[D_i^2 \mid \F_{i-1}]$. Expanding via~\eqref{eq:Di},
\[
\sum_{i=1}^k U_i = A + 2B
\]
where
\[
A := \sum_{p \leq M} c_{pp}\, R_{pp}^{(k)}, \qquad B := \sum_{p < q \leq M} c_{pq}\, R_{pq}^{(k)},
\]
with $c_{pq} := \ln(1 - 1/p) \ln(1 - 1/q)\, \kappa_{pq}$, $\kappa_{pp} := \pi_p q_p$, $\kappa_{pq} := \beta_{\{p,q\}} - q_p q_q$ for $p \neq q$, and
\[
R_{pp}^{(k)} := \sum_{i=1}^k \one\{T_p \geq i\}\, q_p^{2(k-i)}, \qquad R_{pq}^{(k)} := \sum_{i=1}^k \one\{T_p \geq i,\, T_q \geq i\}\, (q_p q_q)^{k-i}.
\]
By Proposition~\ref{prop:disjoint}, $\kappa_{pq} = -\pi_p \pi_q$ whenever $pq > M$.

The main estimates of this section are the following.

\begin{lemma}[Per-prime variance]\label{lem:bd1}
For every $M \geq 51$, every prime $p \leq M$, and every $k \geq 1$,
\begin{equation}\label{eq:Rppbd}
\Var(R_{pp}^{(k)}) \leq \frac{3\, q_p^k}{\pi_p^2}.
\end{equation}
\end{lemma}

\begin{lemma}[Cross-prime covariance]\label{lem:bd2}
There exists a constant $C_0(c) > 0$ (depending only on $c$) such that, for distinct primes $p, q \leq M$ with $pq > M$ and $\pi_p + \pi_q \leq 1/3$, in the joint regime $k = \lfloor cM \rfloor$,
\begin{equation}\label{eq:Covbd}
\bigl|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})\bigr| \leq C_0(c)\, k\, q_p^k q_q^k.
\end{equation}
An admissible value is $C_0(c) = 57 + 54/c$; this is what the proof chain yields directly, though numerical evidence suggests that it is not sharp. Only the shape of the bound --- linear in $k$, multiplicative in $q_p^k q_q^k$ --- is used in Theorem~\ref{thm:HH1} below.
\end{lemma}

\subsection{Proof of Lemma~\ref{lem:bd1}}

Set $Y_p := q_p^{-2(T_p \wedge k)}$. Then $R_{pp}^{(k)} = q_p^{2k}(Y_p - 1)/(1 - q_p^2)$, and
\[
\Var(R_{pp}^{(k)}) = \frac{q_p^{4k}\, \Var(Y_p)}{(1 - q_p^2)^2} \leq \frac{q_p^{4k}\, \EE[Y_p^2]}{(1 - q_p^2)^2}.
\]

Compute
\[
\EE[Y_p^2] = \sum_{m=1}^k q_p^{-4m}\, \pi_p q_p^{m-1} + q_p^{-3k} = \frac{\pi_p}{q_p} \cdot \frac{q_p^{-3k} - 1}{1 - q_p^3} + q_p^{-3k}.
\]
Since $1 - q_p^3 = \pi_p (1 + q_p + q_p^2)$, the first term simplifies to $(q_p^{-3k} - 1)/(q_p (1 + q_p + q_p^2)) \leq q_p^{-3k}/(q_p(1 + q_p + q_p^2))$. Every prime $p \geq 2$ has $m_p \leq M/2$, so $\pi_p \leq (M/2)/(M-1) \leq 51/100$ for $M \geq 51$, and hence $q_p \geq 49/100$ for \emph{every} prime $p \leq M$. Consequently $q_p(1 + q_p + q_p^2) \geq 0.49\,(1 + 0.49 + 0.49^2) > 4/5$, so
\[
\EE[Y_p^2] \leq q_p^{-3k}\bigl(5/4 + 1\bigr) = (9/4)\, q_p^{-3k} \leq 3\, q_p^{-3k}.
\]
Using $(1 - q_p^2)^2 = \pi_p^2 (1 + q_p)^2 \geq \pi_p^2$:
\[
\Var(R_{pp}^{(k)}) \leq \frac{q_p^{4k} \cdot 3 q_p^{-3k}}{\pi_p^2} = \frac{3\, q_p^k}{\pi_p^2}. \qedhere
\]

\subsection{Proof of Lemma~\ref{lem:bd2}}

Write $\PP_{\mathrm{dep}}$ for the joint law of $(T_p, T_q)$ under the true distribution and $\PP_{\mathrm{ind}}$ for the product of the (common) marginals. By Proposition~\ref{prop:disjoint}, at each step three outcomes are possible with probabilities $\beta, \pi_p, \pi_q$ (where $\beta := \beta_{\{p,q\}} = 1 - \pi_p - \pi_q$), giving, for $1 \leq i < j \leq k$,
\[
\PP_{\mathrm{dep}}(T_p = i, T_q = j) = \beta^{i-1}\, \pi_p\, q_q^{j-i-1}\, \pi_q,
\]
and the analogous formula for $j < i$, with $\PP_{\mathrm{dep}}(T_p = T_q < \infty) = 0$ (disjointness). Under $\PP_{\mathrm{dep}}$, the marginal law of $T_p$ is geometric with parameter $\pi_p$ (at each step the event $\{p \mid X\}$ has probability $\pi_p$), and similarly for $T_q$. The product law with identical marginals is then
\[
\PP_{\mathrm{ind}}(T_p = i, T_q = j) = \pi_p q_p^{i-1} \cdot \pi_q q_q^{j-1} = \pi_p \pi_q (q_p q_q)^{i-1} q_q^{j-i} \qquad (i < j),
\]
and analogously for $j < i$ and for the diagonal $\PP_{\mathrm{ind}}(T_p = T_q = m) = \pi_p \pi_q (q_p q_q)^{m-1}$. Set $\delta := \pi_p \pi_q/(q_p q_q)$, so $\beta/(q_p q_q) = 1 - \delta$. Under $\pi_p + \pi_q \leq 1/3$, $q_p q_q \geq (1 - \pi_p)(1 - \pi_q) \geq 1 - (\pi_p + \pi_q) \geq 2/3$, whence $\delta \leq (3/2) \pi_p \pi_q$ and $1/q_p, 1/q_q \leq 3/2$. Since marginals agree, $\Cov_{\mathrm{dep}}(Y_p, Y_q) = \EE_{\mathrm{dep}}[Y_p Y_q] - \EE_{\mathrm{ind}}[Y_p Y_q]$, where $Y_p := q_p^{-2(T_p \wedge k)}$, $Y_q := q_q^{-2(T_q \wedge k)}$.

Partition $\{(i, j) : 1 \leq i, j \leq k+1\}$ (with $T > k$ encoded as $T = k + 1$) into the regions $i < j$, $i > j$, $i = j \leq k$, and $i = j = k+1$.

\textit{Region $i < j$.} The density ratio is
\[
\frac{\PP_{\mathrm{dep}}(T_p = i, T_q = j)}{\PP_{\mathrm{ind}}(T_p = i, T_q = j)} = \frac{(1-\delta)^{i-1}}{q_q}.
\]
Write $(1-\delta)^{i-1}/q_q - 1 = [(1-\delta)^{i-1} - 1]/q_q + \pi_q/q_q$. Using $|(1-\delta)^{i-1} - 1| \leq (i-1)\delta$,
\begin{align*}
&|\EE_{\mathrm{dep}}[Y_p Y_q; i < j]
 - \EE_{\mathrm{ind}}[Y_p Y_q; i < j]|\notag\\
&\qquad\leq (\delta/q_q)\,
\EE_{\mathrm{ind}}[(T_p-1)Y_pY_q; i<j]
 +(\pi_q/q_q)\,\EE_{\mathrm{ind}}[Y_pY_q; i<j].
\end{align*}
Under $\PP_{\mathrm{ind}}$, independence gives $\EE[T_p Y_p Y_q; i < j] \leq \EE[T_p Y_p]\, \EE[Y_q]$. For $r = q_p^{-1} > 1$: $\sum_{m=1}^k m r^m \leq k r^{k+1}/(r-1)$ (telescoping identity $(r-1) \sum_m m r^m = k r^{k+1} - \sum_m r^m$, dropping the positive term). Hence
\[
\EE[T_p Y_p] = \frac{\pi_p}{q_p} \sum_{m=1}^k m q_p^{-m} + (k+1) q_p^{-k} \leq \frac{\pi_p}{q_p} \cdot \frac{k q_p^{-k-1}}{\pi_p} + 2k q_p^{-k} = k q_p^{-k-1} + 2k q_p^{-k} \leq 4k q_p^{-k},
\]
for $q_p \geq 1/2$. Analogously $\EE[Y_q] \leq 3 q_q^{-k}$ (from $\EE[Y_q] = (1 + 1/q_q) q_q^{-k} - 1/q_q \leq 3 q_q^{-k}$) and $\EE[Y_p] \leq 3 q_p^{-k}$. Thus
\[
\EE_{\mathrm{ind}}[T_p Y_p Y_q; i < j] \leq 12k q_p^{-k} q_q^{-k}, \qquad \EE_{\mathrm{ind}}[Y_p Y_q; i<j] \leq 9\, q_p^{-k} q_q^{-k}.
\]
Using $\delta/q_q \leq (3/2)(3/2)\pi_p \pi_q = (9/4)\pi_p \pi_q$ and $\pi_q/q_q \leq (3/2)\pi_q$:
\begin{equation}\label{eq:region-ij}
|\EE_{\mathrm{dep}}[Y_p Y_q; i < j] - \EE_{\mathrm{ind}}[Y_p Y_q; i < j]| \leq 27\, k \pi_p \pi_q\, q_p^{-k} q_q^{-k} + \tfrac{27}{2}\, \pi_q\, q_p^{-k} q_q^{-k}.
\end{equation}

\textit{Region $i > j$.} By symmetry (swap roles of $p$ and $q$):
\begin{equation}\label{eq:region-ji}
|\EE_{\mathrm{dep}}[Y_p Y_q; i > j] - \EE_{\mathrm{ind}}[Y_p Y_q; i > j]| \leq 27\, k \pi_p \pi_q\, q_p^{-k} q_q^{-k} + \tfrac{27}{2}\, \pi_p\, q_p^{-k} q_q^{-k}.
\end{equation}

\textit{Region $i = j \leq k$.} Here $\PP_{\mathrm{dep}} = 0$ and $\PP_{\mathrm{ind}}(T_p = T_q = m) = \pi_p \pi_q (q_p q_q)^{m-1}$, contributing
\[
\EE_{\mathrm{ind}}[Y_p Y_q; T_p = T_q \leq k] = \pi_p \pi_q \sum_{m=1}^{k} (q_p q_q)^{m-1} (q_p q_q)^{-2m} = \pi_p \pi_q (q_p q_q)^{-1} \sum_{m=1}^k (q_p q_q)^{-m},
\]
and using $\sum_{m=1}^k r^m \leq k r^k$ for $r = (q_p q_q)^{-1} \geq 1$,
\begin{equation}\label{eq:region-eq}
\EE_{\mathrm{ind}}[Y_p Y_q; T_p = T_q \leq k] \leq \pi_p \pi_q k (q_p q_q)^{-k-1} \leq \tfrac{3}{2}\, k \pi_p \pi_q\, q_p^{-k} q_q^{-k}.
\end{equation}

\textit{Region $i = j = k+1$.} Here $\PP_{\mathrm{dep}}(T_p > k, T_q > k) = \beta^k$ and $\PP_{\mathrm{ind}}(T_p > k, T_q > k) = (q_p q_q)^k$, with $Y_p Y_q = q_p^{-2k} q_q^{-2k}$:
\[
|\EE_{\mathrm{dep}}[Y_p Y_q; \text{both} > k] - \EE_{\mathrm{ind}}[Y_p Y_q; \text{both} > k]| = (q_p q_q)^{-2k}\bigl|\beta^k - (q_p q_q)^k\bigr|.
\]
Using $|\beta^k - (q_p q_q)^k| = (q_p q_q)^k |1 - (1-\delta)^k| \leq (q_p q_q)^k k \delta$ and $\delta \leq (3/2)\pi_p \pi_q$,
\begin{equation}\label{eq:region-both-tail}
|\EE_{\mathrm{dep}}[Y_p Y_q; \text{both} > k] - \EE_{\mathrm{ind}}[Y_p Y_q; \text{both} > k]| \leq \tfrac{3}{2}\, k \pi_p \pi_q\, (q_p q_q)^{-k} \leq \tfrac{3}{2}\, k \pi_p \pi_q\, q_p^{-k} q_q^{-k}.
\end{equation}

\textit{Assembly.} Combining \eqref{eq:region-ij}--\eqref{eq:region-both-tail},
\begin{equation}\label{eq:Cov-Y}
|\Cov(Y_p, Y_q)| \leq 57\, k \pi_p \pi_q\, q_p^{-k} q_q^{-k} + \tfrac{27}{2}(\pi_p + \pi_q)\, q_p^{-k} q_q^{-k}.
\end{equation}
(The coefficient $57 = 27 + 27 + 3/2 + 3/2$ dominates the individual contributions.) Using the linear relation $R_{pp}^{(k)} = q_p^{2k}(Y_p - 1)/(1 - q_p^2)$ from the proof of Lemma~\ref{lem:bd1} (analogously for $q$) and $(1 - q_p^2)(1 - q_q^2) = \pi_p \pi_q (1 + q_p)(1 + q_q) \geq \pi_p \pi_q$:
\[
|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})| = \frac{(q_p q_q)^{2k}\, |\Cov(Y_p, Y_q)|}{(1 - q_p^2)(1 - q_q^2)} \leq \frac{|\Cov(Y_p, Y_q)|\, (q_p q_q)^{2k}}{\pi_p \pi_q}.
\]
Substituting~\eqref{eq:Cov-Y}:
\[
|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})| \leq \left[57\, k + \tfrac{27}{2}\!\left(\tfrac{1}{\pi_q} + \tfrac{1}{\pi_p}\right)\right] q_p^k q_q^k.
\]
Since $\pi_p \geq 1/(M-1)$ trivially and $k = \lfloor cM \rfloor \geq cM - 1$, we have $M \leq (k+1)/c$, so $1/\pi_p \leq M - 1 \leq (k+1)/c$, and likewise for $q$. Therefore $1/\pi_p + 1/\pi_q \leq 2(k+1)/c$, and using $k+1 \leq 2k$ for $k \geq 1$,
\[
|\Cov(R_{pp}^{(k)}, R_{qq}^{(k)})| \leq \left[57 + 54/c\right] k\, q_p^k q_q^k.
\]
Thus $C_0(c) := 57 + 54/c$ suffices for all $k \geq 1$. \qed

\subsection{Level-summation bound and assembly}

\begin{lemma}[Level-summation upper bound]\label{lem:levelsum}
Let $f \colon \mathbb{N}_{\geq 1} \to \mathbb{R}_{\geq 0}$ satisfy $f(j) \leq A(1 + j)^r$ for $A \geq 0$, $r \geq 0$. For each $c > 0$ there exist $M_0(c, r)$ and $C(c, r)$ such that for $M \geq M_0(c, r)$,
\begin{equation}\label{eq:Lfbd}
L_M(f) := \sum_{p \leq M} \frac{f(m_p)\, q_p^{\lfloor cM \rfloor}}{(p-1)^4} \leq \frac{C(c, r)\, A}{M^3 \ln M}.
\end{equation}
\end{lemma}

\begin{proof}
Partitioning by level and bounding $1/(p - 1)^4 \leq 16(j + 1)^4/M^4$ for $M \geq 2(j + 1)$, together with Lemma~\ref{lem:envelope}:
\[
S_j(M) := \sum_{p \colon m_p = j} \frac{1}{(p-1)^4} \leq \frac{20.16 (j+1)^4}{j \ln(M/j)\, M^3}.
\]
For $j \leq M^{1/2}$, $\ln(M/j) \geq (\ln M)/2$, so $S_j(M) \leq 40.32 (j+1)^4/(j \ln M \cdot M^3)$.

For the exponential factor, $\ln(1 - x) \leq -x$ (concavity) gives $\ln(1 - j/(M-1))^{\lfloor cM \rfloor} \leq -cj + O(j/M)$ uniformly for $j \leq M^{1/2}$, so $(1 - j/(M-1))^{\lfloor cM \rfloor} \leq 2 e^{-cj}$ for $M \geq M_1(c)$.

Main range $j \leq M^{1/2}$:
\[
\sum_{j=1}^{\lfloor M^{1/2} \rfloor} f(j) \cdot (1 - j/(M-1))^{\lfloor cM \rfloor}\, S_j(M) \leq \frac{80.64}{M^3 \ln M} \sum_{j \geq 1} \frac{A (1 + j)^{r+4}\, e^{-cj}}{j},
\]
and the series converges to a finite $\Sigma(c, r) < \infty$.

Tail $j > M^{1/2}$: $q_p^k \leq e^{-c \sqrt{M}/2}$ times $\sum_p 1/(p-1)^4 = O(1)$ times $\sum_{j > M^{1/2}}^{M} A(1 + j)^r = O(A M^{r+1})$, giving super-polynomially small total. Absorbing into the main-range bound yields~\eqref{eq:Lfbd} with $C(c, r) = 81\, \Sigma(c, r)$.
\end{proof}

\begin{theorem}[Conditional-variance concentration]\label{thm:HH1}
As $M \to \infty$ in the joint regime,
\[
\Var\!\left(\sum_{i=1}^k U_i\right)
= O_c\!\left(\frac{1}{M^2(\ln M)^4}\right)
= o(s_k^4).
\]
Consequently $\sum_i U_i/s_k^2 \xrightarrow{P} 1$.
\end{theorem}

\begin{proof}
We have
\[
\Var\!\left(\sum_i U_i\right)=\Var(A)+4\Cov(A,B)+4\Var(B).
\]
Set $J:=M^{1/4}$. Terms involving a prime with $m_p>J$ will be called tail terms. Since
\[
q_p^k\leq \exp\!\left(-\frac{k m_p}{M-1}\right)\leq e^{-cJ/2}
\]
for such primes and all sufficiently large $M$, every tail estimate below is super-polynomially small; polynomial factors in $M$ and powers of $\log\log M$ are therefore harmless.

\textit{Bound on $\Var(A)$.}
By Lemma~\ref{lem:bd1}, which applies to every prime $p \leq M$ once $M \geq 51$,
\[
c_{pp}^2\Var(R_{pp}^{(k)})
\leq \frac{\pi_p^2}{(p-1)^4}\frac{3q_p^k}{\pi_p^2}
=\frac{3q_p^k}{(p-1)^4}.
\]
Lemma~\ref{lem:levelsum} therefore gives
\[
\sum_p c_{pp}^2\Var(R_{pp}^{(k)})
=O_c\!\left(\frac{1}{M^3\ln M}\right).
\]
For the covariance terms with $m_p,m_q\leq J$, we have $pq>M$ and $\pi_p+\pi_q\leq1/3$ for all sufficiently large $M$, so Lemma~\ref{lem:bd2} applies. Hence
\[
\sum_{\substack{p\ne q\\m_p,m_q\leq J}}
|c_{pp}c_{qq}\Cov(R_{pp}^{(k)},R_{qq}^{(k)})|
\leq C_0(c)k\left(\sum_p\frac{\pi_pq_p^k}{(p-1)^2}\right)^2
=O_c\!\left(\frac{1}{M^3(\ln M)^2}\right),
\]
To justify the last estimate explicitly, set
\[
S_M:=\sum_{p\leq M}\frac{\pi_pq_p^k}{(p-1)^2}.
\]
At level $m_p=j\leq M^{1/2}$, use $\pi_p=j/(M-1)$,
$(p-1)^{-2}\leq4(j+1)^2/M^2$, the count in Lemma~\ref{lem:envelope}, and $q_p^k\leq2e^{-cj}$ to obtain
\[
S_M\leq \frac{C}{M^2\ln M}
\sum_{j\geq1}(j+1)^2e^{-cj}
+O(e^{-c\sqrt M/2})
=O_c\!\left(\frac{1}{M^2\ln M}\right).
\]
Since $k=O_c(M)$, this gives $kS_M^2=O_c(M^{-3}(\ln M)^{-2})$ as claimed. If at least one of $m_p,m_q$ exceeds $J$, Cauchy--Schwarz and Lemma~\ref{lem:bd1} give
\[
|c_{pp}|\sqrt{\Var(R_{pp}^{(k)})}
\leq \frac{\sqrt{3}\,q_p^{k/2}}{(p-1)^2},
\]
and the resulting double sum is $O_c(e^{-cJ/4})$. Thus
\begin{equation}\label{eq:VarA-final}
\Var(A)=O_c\!\left(\frac{1}{M^3\ln M}\right).
\end{equation}

\textit{A single-pair bound for $R_{pq}^{(k)}$.}
First suppose $m_p,m_q\leq J$. Then $pq>M$, so the events $\{p\mid X\}$ and $\{q\mid X\}$ are disjoint. Let $\beta=1-\pi_p-\pi_q$, $a=q_pq_q$, and $N=(T_p\wedge T_q)\wedge k$. Since
\[
R_{pq}^{(k)}=\frac{a^k(a^{-N}-1)}{1-a},
\]
we can bound its second moment directly. Put $u:=\pi_p+\pi_q$ and $d:=\pi_p\pi_q$, so that $\beta=1-u$ and $a=1-u+d$. Because $u\leq1/3$ and $d\leq u^2/4$, one has $a^2\leq\beta$ and $1-a=u-d\geq u/2$. Therefore
\begin{align*}
a^{2k}\EE[a^{-2N}]
&=u\sum_{m=1}^k a^{2(k-m)}\beta^{m-1}+\beta^k\\
&\leq ku\beta^{k-1}+\beta^k
\leq C(c)ku\beta^{k-1},
\end{align*}
where the last step uses $ku\geq c/2$ for all sufficiently large $M$. Since $(a^{-N}-1)^2\leq a^{-2N}$,
\begin{equation}\label{eq:Rpq-single}
\Var(R_{pq}^{(k)})
\leq \EE[(R_{pq}^{(k)})^2]
\leq \frac{C(c)k\,\beta^{k-1}}{\pi_p+\pi_q}.
\end{equation}
At a level pair $(j_1, j_2)$ with $j_1, j_2 \leq J$ one has $\pi_p + \pi_q = (j_1+j_2)/(M-1)$, and, since $1 - x \leq e^{-x}$ and $(k-1)/(M-1) = c + O(1/M)$,
\[
\beta^{k-1} \leq \exp\!\left(-\frac{(k-1)(j_1+j_2)}{M-1}\right) \leq 2\, e^{-c(j_1+j_2)}
\]
for $M \geq M_0(c)$, uniformly over $j_1, j_2 \leq J$ (as in~\eqref{eq:uniform-taylor}). Together with $|c_{pq}|\leq \pi_p\pi_q/((p-1)(q-1)) \leq 16(1+j_1)^2(1+j_2)^2/M^4$, the bound~\eqref{eq:Rpq-single} gives
\begin{equation}\label{eq:pair-L2}
|c_{pq}|\sqrt{\Var(R_{pq}^{(k)})}
\leq \frac{C(c)}{M^3}
\frac{(1+j_1)^2(1+j_2)^2}{\sqrt{j_1+j_2}}
 e^{-c(j_1+j_2)/2}.
\end{equation}

\textit{Bound on $\Var(B)$.}
Write $B=B_{\mathrm{main}}+B_{\mathrm{tail}}$, where $B_{\mathrm{main}}$ contains pairs with $m_p,m_q\leq J$. Minkowski's inequality in $L^2$ gives
\[
\sqrt{\Var(B_{\mathrm{main}})}
\leq \sum_{\substack{p<q\\m_p,m_q\leq J}}
|c_{pq}|\sqrt{\Var(R_{pq}^{(k)})}.
\]
By Lemma~\ref{lem:envelope}, the number of primes at level $j\leq J$ is at most $2.52M/(j\ln M)$. Combining this count with~\eqref{eq:pair-L2},
\[
\sqrt{\Var(B_{\mathrm{main}})}
\leq \frac{C(c)}{M(\ln M)^2}
\sum_{j_1,j_2\geq1}
\frac{(1+j_1)^2(1+j_2)^2}{j_1j_2\sqrt{j_1+j_2}}
 e^{-c(j_1+j_2)/2}
=O_c\!\left(\frac{1}{M(\ln M)^2}\right),
\]
because the double series converges.

It remains to control $B_{\mathrm{tail}}$, which also includes every pair with $pq\leq M$. For arbitrary distinct $p,q$, let
\[
\beta_{pq}:=\PP(p\nmid X,q\nmid X),\qquad a:=q_pq_q,
\qquad N:=(T_p\wedge T_q)\wedge k.
\]
If $p$ is chosen so that $m_p=\max(m_p,m_q)$, then $a\leq q_p$ and $\beta_{pq}\leq q_p$. Directly from the geometric law of $T_p\wedge T_q$,
\[
a^{2k}\EE[a^{-2N}]
=(1-\beta_{pq})\sum_{m=1}^k a^{2(k-m)}\beta_{pq}^{m-1}+\beta_{pq}^k
\leq (k+1)q_p^{k-1}.
\]
Since $1-a\geq\pi_p$,
\begin{equation}\label{eq:Rpq-tail}
\Var(R_{pq}^{(k)})
\leq \EE[(R_{pq}^{(k)})^2]
\leq \frac{(k+1)q_p^{k-1}}{\pi_p^2}.
\end{equation}
For every tail pair, $m_p>J$ after relabelling. Using the crude bounds $|\kappa_{pq}|\leq1$, $|\ln(1-1/r)|\leq1/(r-1)$, $1/\pi_p\leq M$, and $\sum_{\substack{r\leq M\\ r\text{ prime}}}(r-1)^{-1}=O(\log\log M)$, Minkowski's inequality and~\eqref{eq:Rpq-tail} give
\[
\sqrt{\Var(B_{\mathrm{tail}})}
\leq C M^{3/2}(\log\log M)^2e^{-cJ/4}
=o(M^{-K})
\]
for every fixed $K$. Therefore
\begin{equation}\label{eq:VarB-final}
\Var(B)=O_c\!\left(\frac{1}{M^2(\ln M)^4}\right).
\end{equation}
This argument controls all covariances among distinct prime-pair summands through Minkowski's inequality; no unproved orthogonality is used.

\textit{Conclusion.}
By Cauchy--Schwarz and~\eqref{eq:VarA-final}--\eqref{eq:VarB-final},
\[
|\Cov(A,B)|
\leq \sqrt{\Var(A)\Var(B)}
=O_c\!\left(\frac{1}{M^{5/2}(\ln M)^{5/2}}\right)
=o\!\left(\frac{1}{M^2(\ln M)^2}\right).
\]
Thus
\[
\Var\!\left(\sum_iU_i\right)
=O_c\!\left(\frac{1}{M^2(\ln M)^4}\right)
=o(s_k^4),
\]
because $s_k^4\asymp M^{-2}(\ln M)^{-2}$ by Theorem~\ref{thm:variance}. Finally,
$\EE[\sum_iU_i]=\sum_i\EE[D_i^2]=s_k^2$, so Chebyshev's inequality gives $\sum_iU_i/s_k^2\xrightarrow{P}1$.
\end{proof}

\begin{lemma}[Linear-scale transfer]\label{lem:linear-transfer}
Let $Y_k:=\varphi(L_k)/L_k$ and $\mu_k:=\EE\ln Y_k$. In the joint regime,
\[
\frac{e^{\mu_k}}{P_M}\longrightarrow1,
\qquad
\frac{Y_k-\rho_k(M)}{P_Ms_k}-\frac{W_k(M)}{s_k}\xrightarrow{P}0.
\]
Consequently the log-scale central limit theorem implies the linear-scale central limit theorem stated in Theorem~\ref{thm:CLT}.
\end{lemma}

\begin{proof}
Writing $a_p:=\ln(1-1/p)<0$,
\[
\mu_k-\ln P_M=-\sum_{p\leq M}a_pq_p^k
=\sum_{p\leq M}|a_p|q_p^k.
\]
Partitioning by $m_p=j$, using $|a_p|\leq1/(p-1)$ and Lemma~\ref{lem:envelope} for $j\leq M^{1/2}$, gives
\[
\sum_{p\leq M}|a_p|q_p^k
\leq \frac{C(c)}{\ln M}\sum_{j\geq1}e^{-cj}+o(1/\ln M)
=O_c(1/\ln M).
\]
The range $j>M^{1/2}$ is super-polynomially small. Hence $e^{\mu_k}/P_M=1+O_c(1/\ln M)$. By Mertens' product theorem~\cite{Tenenbaum2015}, $P_M\asymp1/\ln M$, so also $e^{-\mu_k}=O(\ln M)$.

Since $Y_k=e^{\mu_k+W_k}\leq1$, we have $e^{W_k}\leq e^{-\mu_k}$. For
$R_k:=e^{W_k}-1-W_k$, Taylor's inequality gives
\[
0\leq R_k\leq \tfrac12 e^{(W_k)_+}W_k^2
\leq \tfrac12 e^{-\mu_k}W_k^2.
\]
Therefore
\[
\frac{\EE|R_k|}{s_k}
\leq \frac12 e^{-\mu_k}s_k
=O\!\left(\sqrt{\frac{\ln M}{M}}\right)\longrightarrow0,
\]
where Theorem~\ref{thm:variance} was used. Since $\rho_k(M)=e^{\mu_k}\EE[e^{W_k}]$,
\[
\frac{Y_k-\rho_k(M)}{P_Ms_k}
=\frac{e^{\mu_k}}{P_M}\left(\frac{W_k}{s_k}+\frac{R_k-\EE R_k}{s_k}\right).
\]
The second term in parentheses tends to zero in $L^1$, while $W_k/s_k$ is tight because it has variance one. Together with $e^{\mu_k}/P_M\to1$, this proves the claim. The same estimate also yields the Jensen-gap bound
$0\leq\rho_k(M)-e^{\mu_k}\leq s_k^2/2=o(P_Ms_k)$.
\end{proof}

\section{Proof of Theorem~\ref{thm:CLT}}\label{sec:conclusion-clt}

For each $M$, define the row of the triangular array by
$Z_{M,i}:=D_i/s_k$, $1\leq i\leq k=\lfloor cM\rfloor$, with row filtration $\F_{M,i}:=\F_i$. Theorem~\ref{thm:doob} gives
$\EE[Z_{M,i}\mid\F_{M,i-1}]=0$ and $\sum_iZ_{M,i}=W_k/s_k$. We use the Hall--Heyde martingale central limit theorem~\cite[Thm.~3.2]{HH} in the following standard form: it is enough that, for every $\varepsilon>0$,
\[
\sum_{i=1}^k\EE\!\left[Z_{M,i}^2\one\{|Z_{M,i}|>\varepsilon\}\mid\F_{M,i-1}\right]\xrightarrow{P}0,
\]
and that
\[
\sum_{i=1}^k\EE[Z_{M,i}^2\mid\F_{M,i-1}]\xrightarrow{P}1.
\]
Corollary~\ref{cor:HH2} verifies the first condition and Theorem~\ref{thm:HH1} verifies the second. Hence $W_k/s_k\xrightarrow{d}\NN(0,1)$. Theorem~\ref{thm:variance} and Slutsky's theorem give $\sqrt{M \ln M}\,W_k/\sqrt{B(c)}\xrightarrow{d}\NN(0,1)$. Lemma~\ref{lem:linear-transfer} gives the corresponding linear-scale limit for $\varphi(L_k)/L_k-\rho_k(M)$.
\qed

\section{Discussion}\label{sec:discussion}

\emph{Why the increment bound needs both ingredients.} The trivial Mertens-type bound $|D_i| \leq 2 \Lambda(M) = O(\log \log M)$ is too large to feed into a Lyapunov condition: $(\log \log M)^2 \cdot M \cdot \ln M$ grows, not shrinks. A bound using only the coupon-collector estimate (restricting to primes with $m_p < (\log M)^2$, ignoring the sieve) yields $O(\log \log M)$ contributions per coordinate and is likewise insufficient. A bound using only the sieve-geometry (at most one contributing prime divides $x$) without the coupon-collector filter allows primes with very small $q_p$ (i.e.\ very large $m_p$) that contribute as $O(1/\pi_p) = O(\log \log M)$ to the worst-case $|S_i(x)|$. Both estimates together reduce $|S_i(x)|$ to $(\log M)^2/(M-1)$, with the mean-correction $|C_i|$ contributing an additional $O((\log M)^3/M)$. The resulting uniform bound $\max_i |D_i| = O((\log M)^3/M)$ on $\Omega^*$ is what the Hall--Heyde hypotheses need.

\emph{Finite-$M$ convergence in Theorem~\ref{thm:variance}.} The proof establishes convergence but does not optimise a uniform remainder for the diagonal prime-level sum. The numerical values in Table~\ref{tab:variance} show slow, mildly oscillatory convergence at the tested values of $M$. A quantitative rate would require uniform error estimates in the partial-summation argument of Lemma~\ref{lem:1-over-p-squared}.

\emph{General distributions.} The Doob decomposition (Theorem~\ref{thm:doob}) generalises verbatim to any distribution on $\{2, \ldots, M\}$, with $\xi_p^{(i)}$ redefined as $\one\{p \mid X_i\} - \PP(p \mid X)$. The increment bound uses the uniformity of the $X_i$ in two places: the coupon-collector estimate (which goes through if $\PP(p \mid X) \gtrsim m_p/M$ uniformly) and the sieve-geometric observation (which is distribution-independent once levels are defined by $q_p$). The geometric disjointness in Proposition~\ref{prop:disjoint} is distribution-independent for any law supported on $\{2,\ldots,M\}$: no support point is divisible by both $p$ and $q$ when $pq>M$. What is specific to the uniform law is the explicit formula $\pi_p=m_p/(M-1)$ and the resulting level structure; a non-uniform extension would require distribution-specific estimates for these probabilities and covariances. Extension to product distributions close to uniform appears feasible; substantial departures from uniformity would change the limit $B(c)$.

\emph{Comparison with fixed-$k$ CLT results.} Buraczewski--Iksanov--Marynych~\cite{BIM} established a CLT for $\log \lcm$ of a uniform $m$-tuple on $\{1, \ldots, n\}$ with $m$ fixed as $n \to \infty$. Our result is not comparable in regime: they study $\log \lcm$ itself, with $k$ fixed, while we study $\log(\varphi(L_k)/L_k)$ with $k = \lfloor cM \rfloor$ and $M \to \infty$. The natural common ground --- a CLT that interpolates between the two regimes, for instance $k = f(M) M$ with $f \to 0$ slowly --- is not addressed here.

\section{Conclusion}\label{sec:conclusion}

We have established, in the joint regime $k = \lfloor cM \rfloor$, $M \to \infty$, fixed $c > 0$, the variance asymptotic $M \ln M \cdot s_k^2 \to B(c) = (e^c - 1)^{-1} - (e^{2c} - 1)^{-1}$ (Theorem~\ref{thm:variance}) and a central limit theorem for the standardised log coprime density (Theorem~\ref{thm:CLT}). The variance theorem is proved by a direct diagonal / off-diagonal decomposition, with the off-diagonal term controlled by an exact covariance identity (Lemma~\ref{lem:beta-pq-exact}) valid for prime pairs $(p, q)$ with $pq > M$. The CLT is proved via the Hall--Heyde martingale central limit theorem applied to the Doob filtration. Its two nontrivial ingredients are a uniform increment bound combining coupon-collector concentration with a sieve-geometric observation (Theorem~\ref{thm:supnorm}), and a conditional-variance concentration resting on per-prime moment bounds, the covariance estimate of Lemma~\ref{lem:bd2}, and an $L^2$ Minkowski bound that controls all prime-pair cross terms. Together with the first-order results of the companion paper~\cite{wakid1}, these give a two-moment asymptotic theory for the coprime density in this regime.
